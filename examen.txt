#include <iostream>
#include <cmath>
using namespace std;

class OperacionesBasicas {
private:
    double a, b;

public:
    OperacionesBasicas(double x, double y) {
        a = x;
        b = y;
    }

    double suma() {
        return a + b;
    }

    double resta() {
        return a - b;
    }

    double multiplicacion() {
        return a * b;
    }

    double division() {
        if (b == 0) {
            cout << "Error: No se puede dividir entre 0\n";
            return 0;
        }
        return a / b;
    }
};

class OperacionesAvanzadas {
public:
    double potencia(double base, double exp) {
        return pow(base, exp);
    }

    double porcentaje(double porcentaje, double cantidad) {
        return (porcentaje / 100) * cantidad;
    }

    void derivada(int n) {
        if (n == 0) {
            cout << "f'(x) = 0\n";
        } else if (n == 1) {
            cout << "f'(x) = 1\n";
        } else {
            cout << "f'(x) = " << n << "x^" << (n - 1) << endl;
        }
    }
};
